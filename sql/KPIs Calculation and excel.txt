SELECT salesorders.orddate,
	ROUND(SUM(linenetprice)) AS sales
FROM salesorderdetails, salesorders
WHERE salesorderdetails.orderno = salesorders.orderno
	AND salesorderdetails.completed = 1
	AND salesorders.debtorno LIKE "RETAIL%"
GROUP BY salesorders.orddate
ORDER BY salesorders.orddate DESC

SELECT salesorders.orddate,
	ROUND(SUM(unitprice*(1-discountpercent)* (qtyinvoiced) / currencies.rate)) AS sales
FROM salesorderdetails, salesorders,debtorsmaster,currencies
WHERE salesorderdetails.orderno = salesorders.orderno
	AND debtorsmaster.debtorno = salesorders.debtorno
	AND currencies.currabrev = debtorsmaster.currcode
	AND salesorderdetails.completed = 1
	AND (debtorsmaster.typeid = 9 OR debtorsmaster.typeid = 10)
GROUP BY salesorders.orddate
ORDER BY salesorders.orddate DESC
